    CRITICAL notes from RESEARCH.md:
    - Use `.validator()` NOT `.inputValidator()` (Pitfall 4 — renamed in v1.167)
    - Autocomplete threshold: 0.3 (not 0.6 — Pitfall 6, short names need lower threshold)
    - Use `db.execute(sql\`...\`)` for the pg_trgm `%` operator query (Drizzle ORM doesn't have a direct binding for the `%` operator)

    ```typescript
    import { createServerFn } from '@tanstack/react-start'
    import { z } from 'zod'
    import { sql, and, eq, count } from 'drizzle-orm'
    import { getDb } from '@govtrace/db/client'
    import { entities } from '@govtrace/db/schema/entities'
    import { donations, contracts, grants } from '@govtrace/db/schema/raw'

    // Import lobbyist tables — check actual export names in packages/db/src/schema/raw.ts
    // They may be lobbyRegistrations and lobbyistCommunications

    const SearchInputSchema = z.object({
      query: z.string().min(1).max(200),
      type: z.enum(['all', 'politician', 'company', 'person', 'organization', 'department']).default('all'),
      province: z.string().optional(),
      page: z.number().int().min(1).default(1),
      pageSize: z.number().int().min(1).max(50).default(20),
    })

    const AutocompleteInputSchema = z.object({
      query: z.string().min(1).max(200),
    })

    // Count donations for a list of entity IDs
    async function getEntityCounts(db: ReturnType<typeof getDb>, entityIds: string[]) {
      if (entityIds.length === 0) return {}
      // Use Promise.all for parallel count queries
      // For each entity, count records in each dataset where entityId matches
      // Return map of entityId -> { donations, contracts, grants, lobbying }
      const results: Record<string, { donations: number; contracts: number; grants: number; lobbying: number }> = {}
      for (const id of entityIds) {
        const [donCount, conCount, grCount] = await Promise.all([
          db.select({ c: count() }).from(donations).where(eq(donations.entityId, id)),
          db.select({ c: count() }).from(contracts).where(eq(contracts.entityId, id)),
          db.select({ c: count() }).from(grants).where(eq(grants.entityId, id)),
        ])
        results[id] = {
          donations: Number(donCount[0]?.c ?? 0),
          contracts: Number(conCount[0]?.c ?? 0),
          grants: Number(grCount[0]?.c ?? 0),
          lobbying: 0, // will add lobbyist count if lobby tables have entityId FK
        }
      }
      return results
    }

    export const searchEntities = createServerFn({ method: 'GET' })
      .validator((data: unknown) => SearchInputSchema.parse(data))
      .handler(async ({ data }) => {
        const db = getDb()
        const normalizedQuery = data.query.toLowerCase().trim()

        // Build WHERE clause
        const typeFilter = data.type !== 'all'
          ? sql`AND entity_type = ${data.type}`
          : sql``
        const provinceFilter = data.province
          ? sql`AND province = ${data.province}`
          : sql``

        // pg_trgm similarity search — threshold 0.3 for search results (higher than autocomplete)
        const results = await db.execute<{
          id: string
          canonical_name: string
          entity_type: string
          province: string | null
          score: number
        }>(sql`
          SELECT id, canonical_name, entity_type, province,
                 similarity(normalized_name, ${normalizedQuery}) AS score
          FROM entities
          WHERE normalized_name % ${normalizedQuery}
            AND is_active = true
            ${typeFilter}
            ${provinceFilter}
          ORDER BY score DESC
          LIMIT ${data.pageSize}
          OFFSET ${(data.page - 1) * data.pageSize}
        `)

        const entityIds = results.rows.map((r) => r.id)
        const counts = await getEntityCounts(db, entityIds)

        return {
          results: results.rows.map((r) => ({
            id: r.id,
            canonicalName: r.canonical_name,
            entityType: r.entity_type,
            province: r.province,
            score: r.score,
            counts: counts[r.id] ?? { donations: 0, contracts: 0, grants: 0, lobbying: 0 },
          })),
          page: data.page,
          pageSize: data.pageSize,
        }
      })

    // Lightweight autocomplete — max 8 results, lower threshold, no count queries
    export const getAutocomplete = createServerFn({ method: 'GET' })
      .validator((data: unknown) => AutocompleteInputSchema.parse(data))
      .handler(async ({ data }) => {
        const db = getDb()
        const normalizedQuery = data.query.toLowerCase().trim()

        const results = await db.execute<{
          id: string
          canonical_name: string
          entity_type: string
          score: number
        }>(sql`
          SELECT id, canonical_name, entity_type,
                 similarity(normalized_name, ${normalizedQuery}) AS score
          FROM entities
          WHERE normalized_name % ${normalizedQuery}
            AND is_active = true
          ORDER BY score DESC
          LIMIT 8
        `)

        return results.rows.map((r) => ({
          id: r.id,
          canonicalName: r.canonical_name,
          entityType: r.entity_type,
        }))
      })
    ```

    ```typescript
    import { createAPIFileRoute } from '@tanstack/react-start/api'
    import { searchEntities } from '@/server-fns/search'

    export const APIRoute = createAPIFileRoute('/api/search')({
      GET: async ({ request }) => {
        const url = new URL(request.url)
        const query = url.searchParams.get('q') ?? ''
        const type = url.searchParams.get('type') ?? 'all'
        const province = url.searchParams.get('province') ?? undefined
        const page = Number(url.searchParams.get('page') ?? '1')

        if (!query) {
          return Response.json({ error: 'q parameter is required' }, { status: 400 })
        }

        const results = await searchEntities({
          data: { query, type: type as 'all', province, page },
        })
        return Response.json(results)
      },
    })
    ```
