    Step 3 — Create apps/web/src/components/layout/ThemeProvider.tsx.
    Use cookie-based SSR-safe approach (RESEARCH.md Pattern 6 — NOT next-themes which causes FOUC):

    ```typescript
    // NOTE: No 'use client' directive — not valid in TanStack Start (React 19 RSC model not used here)
    // ThemeProvider is a standard React component rendered in the root layout
    import { createContext, useContext, useState, type ReactNode } from 'react'

    type Theme = 'light' | 'dark'

    const ThemeContext = createContext<{
      theme: Theme
      setTheme: (theme: Theme) => void
    }>({ theme: 'light', setTheme: () => {} })

    export function ThemeProvider({ children, theme: initialTheme }: { children: ReactNode; theme: Theme }) {
      const [theme, setThemeState] = useState<Theme>(initialTheme)

      const setTheme = (newTheme: Theme) => {
        setThemeState(newTheme)
        document.cookie = `theme=${newTheme}; path=/; max-age=31536000; SameSite=Lax`
        document.documentElement.className = newTheme
      }

      return (
        <ThemeContext.Provider value={{ theme, setTheme }}>
          {children}
        </ThemeContext.Provider>
      )
    }

    export function useTheme() {
      return useContext(ThemeContext)
    }
    ```

    Step 4 — Create a server function for theme cookie at apps/web/src/server-fns/theme.ts:

    ```typescript
    import { createServerFn } from '@tanstack/react-start'
    import { getWebRequest } from '@tanstack/react-start/server'

    export const getThemeFn = createServerFn({ method: 'GET' }).handler(async () => {
      const request = getWebRequest()
      const cookieHeader = request?.headers.get('cookie') ?? ''
      const match = /(?:^|;\s*)theme=([^;]*)/.exec(cookieHeader)
      const theme = match?.[1]
      return (theme === 'dark' ? 'dark' : 'light') as 'light' | 'dark'
    })
    ```
