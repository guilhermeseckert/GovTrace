Create an inline sub-tab layout component called `VisualizationsPanel` within the entity route file (not exported, not in a separate file):
```typescript
function VisualizationsPanel({ entityId }: { entityId: string }) {
  const [activeViz, setActiveViz] = useState<'network' | 'moneyFlow' | 'timeline'>('network')

  return (
    <div className="space-y-4">
      {/* Sub-tab selector — styled as button group */}
      <div role="tablist" className="flex gap-2 border-b pb-2">
        {(['network', 'moneyFlow', 'timeline'] as const).map((vizKey) => (
          <button
            key={vizKey}
            role="tab"
            aria-selected={activeViz === vizKey}
            onClick={() => setActiveViz(vizKey)}
            className={`rounded px-3 py-1.5 text-sm font-medium transition-colors
              ${activeViz === vizKey
                ? 'bg-primary text-primary-foreground'
                : 'text-muted-foreground hover:text-foreground'
              }`}
          >
            {en.viz.tabs[vizKey]}
          </button>
        ))}
      </div>

      {/* Visualization panels — render all, hide inactive (preserves D3 zoom state) */}
      <div hidden={activeViz !== 'network'}>
        <NetworkGraph entityId={entityId} />
      </div>
      <div hidden={activeViz !== 'moneyFlow'}>
        <MoneyFlowSankey entityId={entityId} />
      </div>
      <div hidden={activeViz !== 'timeline'}>
        <ActivityTimeline entityId={entityId} />
      </div>
    </div>
  )
}
```

**Important:** Use `hidden` attribute (not conditional rendering) to preserve D3 zoom state when switching tabs — hiding via `display:none` equivalent keeps the SVG element mounted, so d3.zoom listeners remain attached.
