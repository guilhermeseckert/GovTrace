Create `apps/web/src/components/visualizations/shared/useChartColors.ts`:

```typescript
import { useCallback } from 'react'

/**
 * Returns a function that reads a live CSS variable from :root at call time.
 * Using getComputedStyle ensures D3 color values adapt to dark/light mode.
 * Always call getColor() inside the render/useEffect cycle — not at module level.
 */
export function useChartColors() {
  const getColor = useCallback((cssVar: string): string => {
    const raw = getComputedStyle(document.documentElement)
      .getPropertyValue(cssVar)
      .trim()
    // CSS vars in this project are raw HSL values (e.g. "213 54% 23%")
    return raw.startsWith('hsl') ? raw : `hsl(${raw})`
  }, [])

  return {
    getColor,
    nodeColors: {
      politician: () => getColor('--primary'),
      person: () => getColor('--muted-foreground'),
      organization: () => getColor('--accent-foreground'),
      company: () => getColor('--destructive'),
      department: () => getColor('--ring'),
    },
  }
}
```

Create `apps/web/src/components/visualizations/shared/useResizeObserver.ts`:

```typescript
import { useEffect, useState } from 'react'

type Dimensions = { width: number; height: number }

/**
 * Observes a container element's content box size using ResizeObserver.
 * Returns { width: 0, height: 0 } on server and before first measurement.
 * Cleans up the observer on unmount.
 */
export function useResizeObserver(
  ref: React.RefObject<HTMLElement | null>
): Dimensions {
  const [dimensions, setDimensions] = useState<Dimensions>({ width: 0, height: 0 })

  useEffect(() => {
    const el = ref.current
    if (!el) return

    const observer = new ResizeObserver((entries) => {
      const entry = entries[0]
      if (entry) {
        setDimensions({
          width: entry.contentRect.width,
          height: entry.contentRect.height,
        })
      }
    })

    observer.observe(el)
    return () => { observer.disconnect() }
  }, [ref])

  return dimensions
}
```

Create `apps/web/src/server-fns/visualizations.ts`. All three server functions in one file. Use `db.execute(sql`...`)` for recursive CTEs (Drizzle does not support WITH RECURSIVE).

**getGraphData (API-06):**
- Input: `{ id: z.string().uuid(), depth: z.number().int().min(1).max(2).default(1), connectionTypes: z.array(z.string()).optional() }`
- Use the `WITH RECURSIVE ... CYCLE` query from the research doc (Pattern 6). Default depth 1, max depth 2 (not 3 — Pitfall 3 warns depth-3 can explode). Add `LIMIT 150` in the final SELECT.
- After the recursive CTE returns rows, also fetch the root entity itself from `entities` to include as a node at depth 0.
- If `connectionTypes` is provided and non-empty, add a `WHERE connection_type = ANY(${sql.array(connectionTypes)})` clause to the recursive CTE's anchor query.
- Return shape: `{ rootEntityId: string, nodes: Array<{ id, name, entityType, depth }>, edges: Array<{ sourceId, targetId, connectionType, totalValue, transactionCount, firstSeen, lastSeen }>, truncated: boolean }`
- `truncated` is `true` when the raw count of CTE rows exceeded 150 before the LIMIT.

**getMoneyFlow (API-07):**
- Input: `{ id: z.string().uuid() }`
- Do NOT use entity_connections — see Pitfall 7. Query donations + contracts directly.
- Step 1: Select `recipientName, sum(amount) as value` from `donations` WHERE `entityId = id` GROUP BY `recipientName` LIMIT 20 parties.
- Step 2: For each party from step 1, resolve its entity by doing a Drizzle select on `entities` WHERE `canonicalName = recipientName` LIMIT 1.
- Step 3: Select `department, sum(amount) as value` from `contracts` WHERE `entityId IN (party entity IDs from step 2)` GROUP BY `department` LIMIT 20 departments.
- Build nodes: `[{ id: 'root', name: rootEntityName, type: 'entity' }, ...parties as type 'party', ...departments as type 'department']`
- Build links: `[...donationLinks (root→party), ...contractLinks (party→dept)]`
- If total nodes < 3, return `{ nodes: [], links: [] }` (empty state trigger for the Sankey component).

**getTimeline (API-08):**
- Input: `{ id: z.string().uuid() }`
- Use `db.execute(sql`...`)` with a UNION ALL across all 5 datasets. Each branch returns: `date, event_type, description, amount, related_entity_name, source_url`.
  - donations: `date, 'donation', recipientName as description, amount, null as related_entity_name, source_url` WHERE `entityId = id`
  - contracts: `startDate as date, 'contract', department as description, amount, null, source_url` WHERE `entityId = id`
  - grants: `startDate as date, 'grant', department as description, amount, null, source_url` WHERE `entityId = id`
  - lobbyRegistrations: `startDate as date, 'lobby_registration', rawData->>'subject' as description, null, null, null` WHERE `lobbyistEntityId = id OR clientEntityId = id`
  - lobbyCommunications: `date, 'lobby_communication', officialName as description, null, null, null` WHERE `lobbyistEntityId = id OR clientEntityId = id`
- ORDER BY date DESC, LIMIT 500.
- Return shape: `{ events: TimelineEvent[] }` where `TimelineEvent = { date: string, eventType: string, description: string, amount: number | null, relatedEntityName: string | null, sourceUrl: string | null }`.

Type exports: Export `GraphResponse`, `MoneyFlowResponse`, `TimelineResponse` types so visualization components can import them.
