    --- apps/web/src/client.tsx ---
    The client-side hydration entry point. Required by TanStack Start for SSR hydration.

    ```tsx
    import { StartClient } from '@tanstack/react-start'
    import { createRouter } from './router'
    import { hydrateRoot } from 'react-dom/client'
    import { StrictMode } from 'react'

    const router = createRouter()

    hydrateRoot(
      document,
      <StrictMode>
        <StartClient router={router} />
      </StrictMode>,
    )
    ```

    --- apps/web/src/router.tsx ---
    Creates the router instance shared between client and server.

    ```tsx
    import { createRouter as createTanStackRouter } from '@tanstack/react-router'
    import { routeTree } from './routeTree.gen'

    export function createRouter() {
      return createTanStackRouter({
        routeTree,
        scrollRestoration: true,
      })
    }

    declare module '@tanstack/react-router' {
      interface Register {
        router: ReturnType<typeof createRouter>
      }
    }
    ```

    Note: `routeTree.gen` is auto-generated by TanStack Router's Vite plugin on first build. This import will show a TypeScript error until the first `vite build` run — this is expected and correct.

    --- apps/web/src/routes/__root.tsx ---
    The root route wraps all pages with shared HTML shell and metadata.

    ```tsx
    import { createRootRoute, Outlet, ScrollRestoration } from '@tanstack/react-router'
    import { Meta, Scripts } from '@tanstack/react-start'

    export const Route = createRootRoute({
      head: () => ({
        meta: [
          { charSet: 'utf-8' },
          { name: 'viewport', content: 'width=device-width, initial-scale=1' },
          { title: 'GovTrace — Trace the flow of money and influence in Canadian government' },
        ],
        links: [
          { rel: 'icon', href: '/favicon.ico' },
        ],
      }),
      component: RootComponent,
    })

    function RootComponent() {
      return (
        <html lang="en">
          <head>
            <Meta />
          </head>
          <body>
            <Outlet />
            <ScrollRestoration />
            <Scripts />
          </body>
        </html>
      )
    }
    ```
