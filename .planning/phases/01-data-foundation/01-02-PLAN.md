TanStack Start's build output cannot be run with `node dist/server/server.js` directly — it exports a handler, not a runnable server. This is GitHub issue #5476. The fix is a thin wrapper that imports the handler and creates an HTTP server.

**packages/web/server.ts** (custom server wrapper):
```typescript
import { createServer } from 'node:http'
import { createApp, toNodeHandler } from 'srvx'
import { serveStatic } from 'srvx/static'

// Dynamic import to avoid bundling issues with TanStack Start's SSR output
const { default: handler } = await import('./dist/server/server.js')

const app = createApp({
  fetch(req) {
    return handler(req)
  }
})

// Serve static assets from the Vite build output
app.use('/assets', serveStatic({ dir: 'dist/client/assets' }))
app.use('/favicon.ico', serveStatic({ dir: 'dist/client' }))

const port = Number(process.env.PORT ?? 3000)

const server = createServer(toNodeHandler(app))
server.listen(port, () => {
  console.log(`GovTrace web server running on port ${port}`)
})
```

Create the web Dockerfile and update docker-compose.yml to add the web service and fix PostgreSQL port exposure for production use.

**packages/web/Dockerfile**:
```dockerfile
FROM node:20-alpine AS base
RUN npm install -g pnpm@9

# Install dependencies
FROM base AS deps
WORKDIR /app
COPY pnpm-workspace.yaml package.json ./
COPY packages/db/package.json ./packages/db/
COPY packages/web/package.json ./packages/web/
RUN pnpm install --frozen-lockfile

# Build the application
FROM deps AS builder
WORKDIR /app
COPY packages/db ./packages/db
COPY packages/web ./packages/web
# Run database migrations before build
ENV DATABASE_URL=placeholder
RUN pnpm --filter @govtrace/web build
# Compile the server wrapper
RUN pnpm --filter @govtrace/web build:server

# Production image
FROM node:20-alpine AS runner
RUN npm install -g pnpm@9
WORKDIR /app

COPY pnpm-workspace.yaml package.json ./
COPY packages/db/package.json ./packages/db/
COPY packages/web/package.json ./packages/web/
RUN pnpm install --frozen-lockfile --prod

COPY --from=builder /app/packages/web/dist ./packages/web/dist
COPY --from=builder /app/packages/web/server.js ./packages/web/server.js
COPY packages/web/tsconfig.json ./packages/web/

WORKDIR /app/packages/web
EXPOSE 3000

HEALTHCHECK --interval=30s --timeout=10s --start-period=30s --retries=3 \
  CMD wget -qO- http://localhost:3000/health || exit 1

CMD ["node", "server.js"]
```

Replace the existing docker-compose.yml content with:
```yaml
# GovTrace Docker Compose
# Production + Local Dev configuration
#
# LOCAL DEV ONLY — accessing PostgreSQL from host (drizzle-kit studio, etc.):
#   Create docker-compose.override.yml with:
#     services:
#       postgres:
#         ports:
#           - "127.0.0.1:5432:5432"
#   This override is gitignored and only applies locally.
#
# SECURITY NOTE: Never use ports: for postgres in production.
# Docker bypasses UFW and Hetzner firewall rules when ports: is used.
# See: PITFALLS.md — Docker port exposure (INFRA-06)

version: '3.9'

services:
  postgres:
    image: postgres:16-alpine
    environment:
      POSTGRES_USER: govtrace
      POSTGRES_PASSWORD: govtrace
      POSTGRES_DB: govtrace
    # INFRA-06: expose: keeps postgres inside Docker network only
    # Docker bypasses Hetzner firewall with ports: mapping — never use in production
    expose:
      - "5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./docker/postgres-init.sql:/docker-entrypoint-initdb.d/01-extensions.sql:ro
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U govtrace -d govtrace"]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped

  web:
    build:
      context: .
      dockerfile: packages/web/Dockerfile
    environment:
      DATABASE_URL: postgres://govtrace:govtrace@postgres:5432/govtrace
      NODE_ENV: production
      PORT: 3000
    expose:
      - "3000"
    # In Coolify, configure the port mapping via the Coolify UI (proxy handles TLS)
    depends_on:
      postgres:
        condition: service_healthy
    restart: unless-stopped

  ingestion:
    build:
      context: .
      dockerfile: packages/ingestion/Dockerfile
    environment:
      DATABASE_URL: postgres://govtrace:govtrace@postgres:5432/govtrace
      ANTHROPIC_API_KEY: ${ANTHROPIC_API_KEY}
      NODE_ENV: production
    depends_on:
      postgres:
        condition: service_healthy
    profiles:
      - ingestion
    restart: unless-stopped

volumes:
  postgres_data:
```
