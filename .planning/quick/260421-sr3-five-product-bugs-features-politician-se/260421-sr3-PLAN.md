  import type { EntityAggregates } from '@/server-fns/entity-aggregates'

  export function buildFactBlock(entity: EntityProfile, aggregates: EntityAggregates): string {
    const parts: string[] = [`About ${entity.canonicalName}:`]
    if (entity.province) parts.push(`Based in ${entity.province}`)
    parts.push(entity.entityType)
    if (aggregates.earliestYear) parts.push(`First federal record: ${aggregates.earliestYear}`)
    if (aggregates.latestYear && aggregates.latestYear !== aggregates.earliestYear) {
      parts.push(`Most recent: ${aggregates.latestYear}`)
    }
    if (aggregates.largestDeal) {
      const dept = aggregates.largestDeal.department ? ` with ${aggregates.largestDeal.department}` : ''
      parts.push(`Largest single deal: ${formatAmount(String(aggregates.largestDeal.value))}${dept}`)
    }
    return parts.join(' · ')
  }

  export function FactBlock({ entity, aggregates }: { entity: EntityProfile; aggregates: EntityAggregates }) {
    const text = buildFactBlock(entity, aggregates)
    return (
      <div className="rounded-lg border bg-muted/30 p-4 text-sm text-muted-foreground">{text}</div>
    )
  }
